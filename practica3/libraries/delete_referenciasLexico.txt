VALORES QUE CONOZCO QUE SIGNIFICAN 

TOKEN EN -2 SE HA TERMINADO DE ANALIZAR LA CADENA 
TOKEN EN -1 NO HUBO TRANSICION
TOKEN EN 0 NO ES ESTADO DE TRANSICION


DESTINO EN -1 NO HUBO TRANSICION 


INILEXEMA = CARACTER_ACTUAL 
ESTADO_ACTUAL=0

PASANDO POR LOS CARACTERES DE LA CADENA 
    IF(HAY TRANSICION DE ESTADO_ACTUAL C/ CARACTER_ACTUAL)
        ESTADO_ACTUAL LO MUEVO AL OTRO ESTADO 
        IF(ESTADO_ACTUAL ES ESTADO DE ACEPTACION)
            AC = TRUE
            TOKEN = TOKEN DE ESTADO DE ACEPTACION 
            FINLEXEMA = CARACTER_ACTUAL
            LEXEMA.PUSH_BACK(CARACTER_ACTUAL)
        CARACTER ACTUAL LO ACTUALIZO AL SIGUIENTE CARACTER
    ELSE NO HAY TRANSICION 
        IF(AC == TRUE)
            CARACTER_ACTUAL = FINLEXEMA + 1
            INILEXEMA = FINLEXEMA + 1
            --PRACTICAMENTE LEXEMA YA ESTA FORMADA, EN CASO DE QUE SE HAYA PODIDO
            RETURN TOKEN --VAMOS A RETORNAR EL TOKEN QUE YA PREVIAMENTE TENIA
        ELSE NO HAY AC 
            CARACTER_ACTUAL = INILEXEMA + 1
            INILEXEMA = INILEXEMA + 1
            --SE SUPONE QUE LEXEMA NO TIENE NADA
            RETURN TOKEN;


END PASANDO
